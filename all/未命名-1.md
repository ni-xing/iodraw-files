@startuml

class User {
  - id: Long          // 主键ID
  - username: String  // 登录账号
  - password: String  // 加密存储的密码
  - nickname: String  // 昵称
  - avatar: String    // 头像路径（本地）
  - intro: String     // 个人简介
  - createTime: Date  // 创建时间
  + login(username: String, password: String): Boolean  // 登录方法
  + updateInfo(nickname: String, avatar: String, intro: String): void  // 修改个人信息
  + getInfo(): User   // 获取个人信息
}
